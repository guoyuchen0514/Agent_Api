7    import os.path
8    import yaml
9    import logging
10   
11   ERROR = logging.getLogger('error_logger')
12   
13   class GetYamlData:
14       def __init__(self, file_path):
15           self.file_path = file_path
16   
17       def get_yaml_data(self):
18           try:
19               with open(self.file_path, 'r', encoding='utf-8') as f:
20                   data = yaml.safe_load(f)
21               return data
22           except FileNotFoundError as e:
23               ERROR.logger.error("找不到yaml文件：%s", e)
24               raise
25           except Exception as e:
26               ERROR.logger.error("读取yaml文件出错：%s", e)
27               raise
