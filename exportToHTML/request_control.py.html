54       @classmethod
55       def check_headers_str_null(
56               cls,
57               headers: Dict) -> Dict:
58           """ 
59           兼容用户未填写headers或者header值为int 
60           @return: 
61           """
62           headers = ast.literal_eval(cache_regular(str(headers)))
63           if headers is None:
64               headers = {"headers": None}
65           else:
66               for key, value in headers.items():
67                   if not isinstance(value, str):
68                       headers[key] = str(value)
69           return headers

94       def multipart_in_headers(
95               self,
96               request_data:Dict,
97               header : dict):
98           """判断处理header 为 content-type":multipart/form-data"""
99           header =ast.literal_eval(cache_regular(str(header)))
100          request_data =ast.literal_eval(cache_regular(str(request_data)))
101  
102          if header is None:
103              header={"headers":None}
104          else:
105              #将header中的int转换成str
106              for key,value in header.items():
107                  if not isinstance(value,str):
108                      header[key] =str(value)
109              if 'application/json' in str(header.values()):
110                  #判断请求参数不为空，并且参数是字典类型
111                  if request_data and isinstance(request_data,dict):
112                      #当Content-Type为 “multipart/form-data”时，需要将数据类型转换成str
113                      for key,value in request_data.items():
114                          if not isinstance(value ,str ):
115                              request_data[key] =str(value)
116                      request_data =MultipartEncoder(request_data)
117                      header['Content-Type'] =request_data.content_type
118          return request_data,header

145      def upload_file(
146              self) -> Tuple:
147          """ 
148          判断处理上传文件 
149          :return: 
150          """
151          # 处理上传多个文件的情况
152          _files = []
153          file_data = {}
154          # 兼容又要上传文件，又要上传其他类型参数
155          self.file_data_exit(file_data)
156          _data = self.__yaml_case.data
157          for key, value in ast.literal_eval(cache_regular(str(_data)))['file'].items():
158              file_path = ensure_path_sep("\\Files\\" + value)
159              file_data[key] = (value, open(file_path, 'rb'), 'application/octet-stream')
160              _files.append(file_data)
161              # allure中展示该附件
162              allure_attach(source=file_path, name=value, extension=value)
163          multipart = self.multipart_data(file_data)
164          # ast.literal_eval(cache_regular(str(_headers)))['Content-Type'] = multipart.content_type
165          self.__yaml_case.headers['Content-Type'] = multipart.content_type
166          params_data = ast.literal_eval(cache_regular(str(self.file_prams_exit())))
167          return multipart, params_data, self.__yaml_case

189      def request_type_for_params(
190              self,
191              headers: Dict,
192              method: Text,
193              **kwargs):
194  
195          """处理 requestType 为 params """
196          _data = self.__yaml_case.data
197          url = self.__yaml_case.url
198          if _data is not None:
199              # url 拼接的方式传参
200              params_data = "?"
201              for key, value in _data.items():
202                  if value is None or value == '':
203                      params_data += (key + "&")
204                  else:
205                      params_data += (key + "=" + str(value) + "&")
206              url = self.__yaml_case.url + params_data[:-1]
207          _headers = self.check_headers_str_null(headers)
208          res = requests.request(
209              method=method,
210              url=cache_regular(url),
211              headers=_headers,
212              verify=False,
213              data={},
214              params=None,
215              **kwargs)
216          return res

275              self,

284              "detail": yaml_data.detail,

305      @classmethod
306      def get_export_api_filename(cls, res):
307          """ 处理导出文件 """
308          content_disposition = res.headers.get('content-disposition')
309          filename_code = content_disposition.split("=")[-1]  # 分隔字符串，提取文件名
310          filename = urllib.parse.unquote(filename_code)  # url解码
311          return filename

313      def request_type_for_export(
314              self,
315              headers: Dict,
316              method: Text,
317              **kwargs):
318          """判断 requestType 为 export 导出类型"""
319          _headers = self.check_headers_str_null(headers)
320          _data = self.__yaml_case.data
321          _url = self.__yaml_case.url
322          res = requests.request(
323              method=method,
324              url=cache_regular(_url),
325              json=ast.literal_eval(cache_regular(str(_data))),
326              headers=_headers,
327              verify=False,
328              stream=False,
329              data={},
330              **kwargs)
331          filepath = os.path.join(ensure_path_sep("\\Files\\"), self.get_export_api_filename(res))  # 拼接路径
332          if res.status_code == 200:
333              if res.text:  # 判断文件内容是否为空
334                  with open(filepath, 'wb') as file:
335                      # iter_content循环读取信息写入，chunk_size设置文件大小
336                      for chunk in res.iter_content(chunk_size=1):
337                          file.write(chunk)
338              else:
339                  print("文件为空")
340  
341          return res

376      @log_decorator(True)
377      @execution_duration(3000)
378      # @encryption("md5")
379      def http_request(
380              self,
381              dependent_switch=True,
382              **kwargs
383      ):
384          from utils.requests_tool.dependent_case import DependentCase
385          requests_type_mapping = {
386              RequestType.JSON.value: self.request_type_for_json,
387              RequestType.DATA.value: self.request_type_for_data,
388              RequestType.PARAMS.value: self.request_type_for_params,
389              RequestType.FILE.value: self.request_type_for_file,
390              RequestType.NONE.value: self.request_type_for_none,
391              RequestType.EXPORT.value: self.request_type_for_export
392          }
393          is_run = ast.literal_eval(cache_regular(str(self.__yaml_case.is_run)))
394          # 判断用例是否执行
395          if is_run is True or is_run is None:
396              # 处理多业务逻辑
397              if dependent_switch is True:
398                   DependentCase(self.__yaml_case).get_dependent_data()
399  
400              """ 
401              调用mapping里面的值 
402              """
403              res = requests_type_mapping.get(self.__yaml_case.requestType)(
404                  headers=self.__yaml_case.headers,
405                  method=self.__yaml_case.method,
406                  **kwargs
407              )
408              if self.__yaml_case.sleep is not None:
409                  time.sleep(self.__yaml_case.sleep)
410  
411              _res_data = self._check_params(
412                  res=res,
413                  yaml_data=self.__yaml_case)
414  
415              self.api_allure_step(
416                  url=_res_data.url,
417                  headers=str(_res_data.headers),
418                  method=_res_data.method,
419                  data=str(_res_data.body),
420                  assert_data=str(_res_data.assert_data),
421                  res_time=str(_res_data.res_time),
422                  res=_res_data.response_data
423              )
424  
425              # 将当前请求数据存入缓存中
426              SetCurrentRequestCache(
427                  current_request_set_cache=self.__yaml_case.current_request_set_cache,
428                  request_data=self.__yaml_case.data,
429                  response_data=res
430              ).set_caches_main()
431  
432              return _res_data
