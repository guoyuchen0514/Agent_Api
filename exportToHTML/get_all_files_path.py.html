9    import os
10   
11   def get_all_files(file_path, yaml_data_switch=False) -> list:
12       filename = []
13       for root, dirs, files in os.walk(file_path):
14           for file_name in files:
15               path = os.path.join(root, file_name)
16               if yaml_data_switch:
17                   if file_name.endswith(".yaml") or file_name.endswith(".yml"):
18                       filename.append(path)
19               else:
20                   filename.append(path)
21       return filename
